    –ê–∫—Å–∏–æ–º—ã —Ç–µ–æ—Ä–∏–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π. –î–∏—Å–∫—Ä–µ—Ç–Ω–∞—è –∏ –Ω–µ–ø—Ä–µ—Ä—ã–≤–Ω–∞—è —Å–ª—É—á–∞–π–Ω—ã–µ –≤–µ–ª–∏—á–∏–Ω—ã.¶

    –¢–µ–æ—Ä–∏—è –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π —è–≤–ª—è–µ—Ç—Å—è –≤–∞–∂–Ω—ã–º —Ä–∞–∑–¥–µ–ª–æ–º –º–∞—Ç–µ–º–∞—Ç–∏–∫–∏, –∏–∑—É—á–∞—é—â–∏–º —Å–ª—É—á–∞–π–Ω—ã–µ —Å–æ–±—ã—Ç–∏—è –∏ –∏—Ö –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏. –û—Å–Ω–æ–≤—É —ç—Ç–æ–π —Ç–µ–æ—Ä–∏–∏ —Å–æ—Å—Ç–∞–≤–ª—è—é—Ç –∞–∫—Å–∏–æ–º—ã –ö–æ–ª–º–æ–≥–æ—Ä–æ–≤–∞, –Ω–∞ –∫–æ—Ç–æ—Ä—ã—Ö —Å—Ç—Ä–æ–∏—Ç—Å—è –≤—Å—è –æ—Å—Ç–∞–ª—å–Ω–∞—è —Å—Ç—Ä—É–∫—Ç—É—Ä–∞. –¢–∞–∫–∂–µ –≤ —Ç–µ–æ—Ä–∏–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π —Ä–∞–∑–ª–∏—á–∞—é—Ç –¥–∏—Å–∫—Ä–µ—Ç–Ω—ã–µ –∏ –Ω–µ–ø—Ä–µ—Ä—ã–≤–Ω—ã–µ —Å–ª—É—á–∞–π–Ω—ã–µ –≤–µ–ª–∏—á–∏–Ω—ã, –∫–∞–∂–¥–∞—è –∏–∑ –∫–æ—Ç–æ—Ä—ã—Ö –∏–º–µ–µ—Ç —Å–≤–æ–∏ –æ—Å–æ–±–µ–Ω–Ω–æ—Å—Ç–∏. –†–∞—Å—Å–º–æ—Ç—Ä–∏–º —ç—Ç–∏ –ø–æ–Ω—è—Ç–∏—è –±–æ–ª–µ–µ –ø–æ–¥—Ä–æ–±–Ω–æ.

    –ê–∫—Å–∏–æ–º—ã —Ç–µ–æ—Ä–∏–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π¶

    –ê–∫—Å–∏–æ–º–∞—Ç–∏—á–µ—Å–∫–∏–π –ø–æ–¥—Ö–æ–¥ –∫ —Ç–µ–æ—Ä–∏–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π –±—ã–ª —Ä–∞–∑—Ä–∞–±–æ—Ç–∞–Ω —Å–æ–≤–µ—Ç—Å–∫–∏–º –º–∞—Ç–µ–º–∞—Ç–∏–∫–æ–º –ê–Ω–¥—Ä–µ–µ–º –ù–∏–∫–æ–ª–∞–µ–≤–∏—á–µ–º –ö–æ–ª–º–æ–≥–æ—Ä–æ–≤—ã–º –≤ 1933 –≥–æ–¥—É. –û—Å–Ω–æ–≤–Ω—ã–µ –∞–∫—Å–∏–æ–º—ã —Ç–µ–æ—Ä–∏–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π —Å–ª–µ–¥—É—é—â–∏–µ:

    1. –ê–∫—Å–∏–æ–º–∞ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏: –î–ª—è –ª—é–±–æ–≥–æ —Å–æ–±—ã—Ç–∏—è \( A \) –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å \( P(A) \) –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞. \(P(A) \geq 0\)

    2. –ê–∫—Å–∏–æ–º–∞ –Ω–æ—Ä–º–∏—Ä–æ–≤–∫–∏: –í–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å –¥–æ—Å—Ç–æ–≤–µ—Ä–Ω–æ–≥–æ —Å–æ–±—ã—Ç–∏—è (—Ç–æ –µ—Å—Ç—å —Å–æ–±—ã—Ç–∏—è, –∫–æ—Ç–æ—Ä–æ–µ –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç –≤—Å–µ–≥–¥–∞) —Ä–∞–≤–Ω–∞ 1. \(P(\Omega) = 1\) –ó–¥–µ—Å—å \( \Omega \) ‚Äî –º–Ω–æ–∂–µ—Å—Ç–≤–æ –≤—Å–µ—Ö –≤–æ–∑–º–æ–∂–Ω—ã—Ö –∏—Å—Ö–æ–¥–æ–≤ (–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–æ —ç–ª–µ–º–µ–Ω—Ç–∞—Ä–Ω—ã—Ö –∏—Å—Ö–æ–¥–æ–≤).

    3. –ê–∫—Å–∏–æ–º–∞ –∞–¥–¥–∏—Ç–∏–≤–Ω–æ—Å—Ç–∏: –ï—Å–ª–∏ —Å–æ–±—ã—Ç–∏—è \( A \) –∏ \( B \) –Ω–µ—Å–æ–≤–º–µ—Å—Ç–Ω—ã (—Ç–æ –µ—Å—Ç—å –æ–Ω–∏ –Ω–µ –º–æ–≥—É—Ç –ø—Ä–æ–∏–∑–æ–π—Ç–∏ –æ–¥–Ω–æ–≤—Ä–µ–º–µ–Ω–Ω–æ), —Ç–æ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—è —ç—Ç–∏—Ö —Å–æ–±—ã—Ç–∏–π —Ä–∞–≤–Ω–∞ —Å—É–º–º–µ –∏—Ö –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π. \(P(A \cup B) = P(A) + P(B), \quad \text{–µ—Å–ª–∏} \quad A \cap B = \emptyset\) \(P(A \cup B) = P(A) + P(B) - P(A \cap B), \quad \text{–µ—Å–ª–∏} \quad A \cap B \ne \emptyset\)

    –ù–∞ –æ—Å–Ω–æ–≤–µ —ç—Ç–∏—Ö –∞–∫—Å–∏–æ–º –º–æ–∂–Ω–æ –≤—ã–≤–µ—Å—Ç–∏ –º–Ω–æ–∂–µ—Å—Ç–≤–æ –¥—Ä—É–≥–∏—Ö —Å–≤–æ–π—Å—Ç–≤ –∏ —Ñ–æ—Ä–º—É–ª —Ç–µ–æ—Ä–∏–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π.

    –î–∏—Å–∫—Ä–µ—Ç–Ω—ã–µ —Å–ª—É—á–∞–π–Ω—ã–µ –≤–µ–ª–∏—á–∏–Ω—ã¶

    –î–∏—Å–∫—Ä–µ—Ç–Ω–∞—è —Å–ª—É—á–∞–π–Ω–∞—è –≤–µ–ª–∏—á–∏–Ω–∞ –ø—Ä–∏–Ω–∏–º–∞–µ—Ç –æ—Ç–¥–µ–ª—å–Ω—ã–µ, —á—ë—Ç–∫–æ –æ–ø—Ä–µ–¥–µ–ª—ë–Ω–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è. –ü—Ä–∏–º–µ—Ä–æ–º –º–æ–∂–µ—Ç —Å–ª—É–∂–∏—Ç—å –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –≤—ã–ø–∞–≤—à–∏—Ö –æ—á–∫–æ–≤ –ø—Ä–∏ –±—Ä–æ—Å–∫–µ –∫—É–±–∏–∫–∞. –û—Å–Ω–æ–≤–Ω—ã–µ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–∏ –¥–∏—Å–∫—Ä–µ—Ç–Ω—ã—Ö —Å–ª—É—á–∞–π–Ω—ã—Ö –≤–µ–ª–∏—á–∏–Ω:

    • –§—É–Ω–∫—Ü–∏—è —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π (Probability Mass Function, PMF): –û–ø—Ä–µ–¥–µ–ª—è–µ—Ç –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å –∫–∞–∂–¥–æ–≥–æ –≤–æ–∑–º–æ–∂–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è —Å–ª—É—á–∞–π–Ω–æ–π –≤–µ–ª–∏—á–∏–Ω—ã. \(P(X = x_i) = p_i\) –≥–¥–µ \(x_i\) ‚Äî –≤–æ–∑–º–æ–∂–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è —Å–ª—É—á–∞–π–Ω–æ–π –≤–µ–ª–∏—á–∏–Ω—ã \(X\), –∞ \(p_i\) ‚Äî —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–µ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏.

    • –ú–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ (–°—Ä–µ–¥–Ω–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ): \(\mathbb{E}[X] = \sum_{i} x_i \cdot P(X = x_i)\)

    • –î–∏—Å–ø–µ—Ä—Å–∏—è: –ú–µ—Ä–∞ —Ä–∞–∑–±—Ä–æ—Å–∞ –∑–Ω–∞—á–µ–Ω–∏–π —Å–ª—É—á–∞–π–Ω–æ–π –≤–µ–ª–∏—á–∏–Ω—ã –≤–æ–∫—Ä—É–≥ –µ—ë —Å—Ä–µ–¥–Ω–µ–≥–æ. \(\text{Var}(X) = \mathbb{E}[(X - \mathbb{E}[X])^2] = \sum_{i} (x_i - \mathbb{E}[X])^2 \cdot P(X = x_i)\)

    –ù–µ–ø—Ä–µ—Ä—ã–≤–Ω—ã–µ —Å–ª—É—á–∞–π–Ω—ã–µ –≤–µ–ª–∏—á–∏–Ω—ã¶

    –ù–µ–ø—Ä–µ—Ä—ã–≤–Ω–∞—è —Å–ª—É—á–∞–π–Ω–∞—è –≤–µ–ª–∏—á–∏–Ω–∞ –º–æ–∂–µ—Ç –ø—Ä–∏–Ω–∏–º–∞—Ç—å –ª—é–±—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è –∏–∑ –Ω–µ–∫–æ—Ç–æ—Ä–æ–≥–æ –∏–Ω—Ç–µ—Ä–≤–∞–ª–∞ –∏–ª–∏ –º–Ω–æ–∂–µ—Å—Ç–≤–∞. –ü—Ä–∏–º–µ—Ä–æ–º –º–æ–∂–µ—Ç —Å–ª—É–∂–∏—Ç—å –∏–∑–º–µ—Ä–µ–Ω–∏–µ —Ä–æ—Å—Ç–∞ —á–µ–ª–æ–≤–µ–∫–∞. –û—Å–Ω–æ–≤–Ω—ã–µ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–∏ –Ω–µ–ø—Ä–µ—Ä—ã–≤–Ω—ã—Ö —Å–ª—É—á–∞–π–Ω—ã—Ö –≤–µ–ª–∏—á–∏–Ω:

    • –§—É–Ω–∫—Ü–∏—è –ø–ª–æ—Ç–Ω–æ—Å—Ç–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ (Probability Density Function, PDF): –û–ø—Ä–µ–¥–µ–ª—è–µ—Ç –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å —Ç–æ–≥–æ, —á—Ç–æ —Å–ª—É—á–∞–π–Ω–∞—è –≤–µ–ª–∏—á–∏–Ω–∞ –ø–æ–ø–∞–¥—ë—Ç –≤ –º–∞–ª—ã–π –∏–Ω—Ç–µ—Ä–≤–∞–ª –≤–æ–∫—Ä—É–≥ –¥–∞–Ω–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è. \(P(a \leq X \leq b) = \int_{a}^{b} f(x) \, dx\) –≥–¥–µ \(f(x)\) ‚Äî —Ñ—É–Ω–∫—Ü–∏—è –ø–ª–æ—Ç–Ω–æ—Å—Ç–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏.

    • –§—É–Ω–∫—Ü–∏—è —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è (Cumulative Distribution Function, CDF): –û–ø—Ä–µ–¥–µ–ª—è–µ—Ç –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å —Ç–æ–≥–æ, —á—Ç–æ —Å–ª—É—á–∞–π–Ω–∞—è –≤–µ–ª–∏—á–∏–Ω–∞ –ø—Ä–∏–º–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ –º–µ–Ω—å—à–µ –∏–ª–∏ —Ä–∞–≤–Ω–æ–µ –¥–∞–Ω–Ω–æ–º—É. \(F(x) = P(X \leq x) = \int_{-\infty}^{x} f(t) \, dt\)

    • –ú–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ: \(\mathbb{E}[X] = \int_{-\infty}^{\infty} x f(x) \, dx\)

    • –î–∏—Å–ø–µ—Ä—Å–∏—è: \(\text{Var}(X) = \mathbb{E}[(X - \mathbb{E}[X])^2] = \int_{-\infty}^{\infty} (x - \mathbb{–ï}[X])^2 f(x) \, dx\)

    import numpy as np
import scipy.stats as stats

class DiscreteRandomVariable:
    def __init__(self, outcomes, probabilities):
        self.outcomes = outcomes
        self.probabilities = probabilities

    def mean(self):
        return np.sum(self.outcomes * self.probabilities)

    def variance(self):
        mean = self.mean()
        return np.sum((self.outcomes - mean) ** 2 * self.probabilities)

    @classmethod
    def from_pmf(cls, pmf):
        outcomes, probabilities = zip(*pmf.items())
        return cls(np.array(outcomes), np.array(probabilities))

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è –¥–ª—è –±—Ä–æ—Å–∫–∞ –∫—É–±–∏–∫–∞
outcomes = np.array([1, 2, 3, 4, 5, 6])
probabilities = np.ones(6) / 6  # –∫–∞–∂–¥—ã–π –∏—Å—Ö–æ–¥ —Ä–∞–≤–Ω–æ–≤–µ—Ä–æ—è—Ç–µ–Ω
dice = DiscreteRandomVariable(outcomes, probabilities)
print("–°—Ä–µ–¥–Ω–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ (–ú–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ) –¥–ª—è –±—Ä–æ—Å–∫–∞ –∫—É–±–∏–∫–∞:", dice.mean())
print("–î–∏—Å–ø–µ—Ä—Å–∏—è –¥–ª—è –±—Ä–æ—Å–∫–∞ –∫—É–±–∏–∫–∞:", dice.variance())

class ContinuousRandomVariable:
    def __init__(self, distribution):
        self.distribution = distribution

    def mean(self):
        return self.distribution.mean()

    def variance(self):
        return self.distribution.var()

    @classmethod
    def from_pdf(cls, pdf, lower_bound, upper_bound):
        distribution = stats.rv_continuous()
        distribution.pdf = pdf
        distribution.a = lower_bound
        distribution.b = upper_bound
        return cls(distribution)

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è –¥–ª—è –Ω–æ—Ä–º–∞–ª—å–Ω–æ–≥–æ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è
normal_pdf = lambda x: stats.norm.pdf(x, loc=0, scale=1)
normal_distribution = ContinuousRandomVariable.from_pdf(normal_pdf, -np.inf, np.inf)
print("–°—Ä–µ–¥–Ω–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ (–ú–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ) –¥–ª—è –Ω–æ—Ä–º–∞–ª—å–Ω–æ–≥–æ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è:", normal_distribution.mean())
print("–î–∏—Å–ø–µ—Ä—Å–∏—è –¥–ª—è –Ω–æ—Ä–º–∞–ª—å–Ω–æ–≥–æ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è:", normal_distribution.variance())
