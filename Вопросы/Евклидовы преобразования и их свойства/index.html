    1. –°–º–µ—â–µ–Ω–∏–µ (–¢—Ä–∞–Ω—Å–ª—è—Ü–∏—è)¶

    –ú–∞—Ç–µ–º–∞—Ç–∏–∫–∞: –°–º–µ—â–µ–Ω–∏–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ—Ç —Å–æ–±–æ–π –¥–æ–±–∞–≤–ª–µ–Ω–∏–µ –ø–æ—Å—Ç–æ—è–Ω–Ω–æ–≥–æ –≤–µ–∫—Ç–æ—Ä–∞ –∫–æ –≤—Å–µ–º —Ç–æ—á–∫–∞–º –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞. –ï—Å–ª–∏ —Ç–æ—á–∫–∞ \(\mathbf{v}\) –ø–µ—Ä–µ–º–µ—â–∞–µ—Ç—Å—è –Ω–∞ –≤–µ–∫—Ç–æ—Ä \(\mathbf{t}\), —Ç–æ –Ω–æ–≤–æ–µ –ø–æ–ª–æ–∂–µ–Ω–∏–µ —Ç–æ—á–∫–∏ \(\mathbf{v}'\) –±—É–¥–µ—Ç: \(\mathbf{v}' = \mathbf{v} + \mathbf{t}\)

    –ö–æ–¥ –Ω–∞ Python: 

    import numpy as np

# –ò—Å—Ö–æ–¥–Ω–∞—è —Ç–æ—á–∫–∞
v = np.array([1, 2, 3])
# –í–µ–∫—Ç–æ—Ä —Å–º–µ—â–µ–Ω–∏—è
t = np.array([4, 5, 6])

# –ù–æ–≤–∞—è —Ç–æ—á–∫–∞ –ø–æ—Å–ª–µ —Å–º–µ—â–µ–Ω–∏—è
v_prime = v + t
print(v_prime)
# –†–µ–∑—É–ª—å—Ç–∞—Ç: [5 7 9]

    2. –ü–æ–≤–æ—Ä–æ—Ç (–†–æ—Ç–∞—Ü–∏—è)¶

    –ú–∞—Ç–µ–º–∞—Ç–∏–∫–∞: –ü–æ–≤–æ—Ä–æ—Ç —Ç–æ—á–∫–∏ –≤ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ –Ω–∞ —É–≥–æ–ª \(\theta\) –≤–æ–∫—Ä—É–≥ –∑–∞–¥–∞–Ω–Ω–æ–π –æ—Å–∏ \(\mathbf{u}\) (–Ω–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω–æ–π) –º–æ–∂–Ω–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–∏—Ç—å –º–∞—Ç—Ä–∏—Ü–µ–π –ø–æ–≤–æ—Ä–æ—Ç–∞ \(\mathbf{R}\). –î–ª—è –¥–≤—É–º–µ—Ä–Ω–æ–≥–æ —Å–ª—É—á–∞—è –º–∞—Ç—Ä–∏—Ü–∞ –ø–æ–≤–æ—Ä–æ—Ç–∞ –Ω–∞ —É–≥–æ–ª \(\theta\) –≤—ã–≥–ª—è–¥–∏—Ç —Ç–∞–∫: \(\mathbf{R} = \begin{pmatrix} \cos \theta & -\sin \theta \\ \sin \theta & \cos \theta \end{pmatrix}\)

    –ö–æ–¥ –Ω–∞ Python (–¥–ª—è –¥–≤—É–º–µ—Ä–Ω–æ–≥–æ —Å–ª—É—á–∞—è): 

    import numpy as np

# –£–≥–æ–ª –ø–æ–≤–æ—Ä–æ—Ç–∞ –≤ —Ä–∞–¥–∏–∞–Ω–∞—Ö
theta = np.radians(45)
# –ú–∞—Ç—Ä–∏—Ü–∞ –ø–æ–≤–æ—Ä–æ—Ç–∞
R = np.array([[np.cos(theta), -np.sin(theta)], [np.sin(theta), np.cos(theta)]])

# –ò—Å—Ö–æ–¥–Ω–∞—è —Ç–æ—á–∫–∞
v = np.array([1, 0])

# –ù–æ–≤–∞—è —Ç–æ—á–∫–∞ –ø–æ—Å–ª–µ –ø–æ–≤–æ—Ä–æ—Ç–∞
v_prime = np.dot(R, v)
print(v_prime)
# –†–µ–∑—É–ª—å—Ç–∞—Ç: [0.70710678 0.70710678] (–ø—Ä–∏–º–µ—Ä–Ω–æ [sqrt(2)/2, sqrt(2)/2])

    3. –û—Ç—Ä–∞–∂–µ–Ω–∏–µ (–†–µ—Ñ–ª–µ–∫—Å–∏—è)¶

    –ú–∞—Ç–µ–º–∞—Ç–∏–∫–∞: –û—Ç—Ä–∞–∂–µ–Ω–∏–µ —Ç–æ—á–∫–∏ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –æ—Å–∏ –º–æ–∂–Ω–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–∏—Ç—å –º–∞—Ç—Ä–∏—Ü–µ–π –æ—Ç—Ä–∞–∂–µ–Ω–∏—è. –ù–∞–ø—Ä–∏–º–µ—Ä, –¥–ª—è –æ—Ç—Ä–∞–∂–µ–Ω–∏—è –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –æ—Å–∏ \(x\) –≤ –¥–≤—É–º–µ—Ä–Ω–æ–º –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –º–∞—Ç—Ä–∏—Ü–∞: \(\mathbf{M_x} = \begin{pmatrix} 1 & 0 \\ 0 & -1 \end{pmatrix}\)

    –ö–æ–¥ –Ω–∞ Python (–¥–ª—è –¥–≤—É–º–µ—Ä–Ω–æ–≥–æ —Å–ª—É—á–∞—è): 

    import numpy as np

# –ú–∞—Ç—Ä–∏—Ü–∞ –æ—Ç—Ä–∞–∂–µ–Ω–∏—è –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –æ—Å–∏ x
M_x = np.array([[1, 0], [0, -1]])

# –ò—Å—Ö–æ–¥–Ω–∞—è —Ç–æ—á–∫–∞
v = np.array([1, 2])

# –ù–æ–≤–∞—è —Ç–æ—á–∫–∞ –ø–æ—Å–ª–µ –æ—Ç—Ä–∞–∂–µ–Ω–∏—è
v_prime = np.dot(M_x, v)
print(v_prime)
# –†–µ–∑—É–ª—å—Ç–∞—Ç: [ 1 -2]

    1. –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–π: –î–ª—è –ª—é–±—ã—Ö –¥–≤—É—Ö —Ç–æ—á–µ–∫ \(\mathbf{v}_1\) –∏ \(\mathbf{v}_2\) –≤ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –º–µ–∂–¥—É –Ω–∏–º–∏ –æ—Å—Ç–∞–µ—Ç—Å—è –Ω–µ–∏–∑–º–µ–Ω–Ω—ã–º –ø–æ—Å–ª–µ –ø—Ä–∏–º–µ–Ω–µ–Ω–∏—è –ª—é–±–æ–≥–æ –µ–≤–∫–ª–∏–¥–æ–≤–∞ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è.

    –ü—Ä–∏–º–µ—Ä –ø—Ä–æ–≤–µ—Ä–∫–∏ —Å–≤–æ–π—Å—Ç–≤–∞: 

    import numpy as np

# –ò—Å—Ö–æ–¥–Ω—ã–µ —Ç–æ—á–∫–∏
v1 = np.array([1, 2])
v2 = np.array([4, 6])
distance_before = np.linalg.norm(v1 - v2)

# –í–µ–∫—Ç–æ—Ä —Å–º–µ—â–µ–Ω–∏—è
t = np.array([3, 3])
v1_prime = v1 + t

distance_after = np.linalg.norm(v1_prime - v2_prime)

print(distance_before)  # –†–µ–∑—É–ª—å—Ç–∞—Ç: 5.0
print(distance_after)   # –†–µ–∑—É–ª—å—Ç–∞—Ç: 5.0
